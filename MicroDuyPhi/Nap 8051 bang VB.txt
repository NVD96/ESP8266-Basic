/*********Hai hàm SPI cơ bản***********
void SPISendByte(unsigned char byte) {
unsigned char i;
DDRB.5 = 1;
DDRB.7 = 1;
SCK = 0;
delay_us(10);
for (i=0;i<=7;i++) {
if (byte & 0x80) MOSI = 1;
else MOSI = 0;
byte = byte << 1;
SCK = 1;
delay_us(10);
SCK = 0;
delay_us(10);
}
MOSI = 0;
delay_us(10);
}
//******************************
unsigned char SPIGetByte() {
unsigned char i,dulieu,bitdulieu;
dulieu = 0;
MISO = 1;
DDRB.6 = 0;
DDRB.7 = 1;
SCK = 0;
delay_us(10);
for (i=0;i<=7;i++) {
SCK = 1;
delay_us(10);
if (PINB & 0x40) bitdulieu = 0x01;
else bitdulieu = 0x00;
dulieu = dulieu << 1;
dulieu = dulieu | bitdulieu;
SCK = 0;
delay_us(10);
}
MISO = 0;
delay_us(10);
return dulieu;
}
